# numpy求解线性代数基本问题

## 输入

```python
import numpy as np

a = np.array([[1,2],[3,4]])
print(a,"\n","-"*100)
#对array转置
print(a.transpose(),"\n","-"*100)
#用T一样的效果，而且T是python原生自带
print(a.T,"\n","-"*100)
#矩阵求逆
print(np.linalg.inv(a),"\n","-"*100)
#生成单位矩阵
print(np.eye(3),"\n","-"*100)
#矩阵乘法,dot函数
print(np.dot(a,a),"\n","-"*100)
#矩阵的迹
print(np.trace(a),"\n","-"*100)
#求解线性方程组
y = np.array([[5],[7]])
print(np.linalg.solve(a, y),"\n","-"*100)

```

# ix_()函数

## 输入

```python
import numpy as np

a = np.array([2,3,4,5])

b = np.array([8,5,4])

c = np.array([5,4,6,8,3])

#The ix_ function can be used to combine \
#different vectors so as to obtain the result for each n-uplet
ax,bx,cx = np.ix_(a,b,c)
#ix_函数可以为了获得多元组的结果而用来结合不同向量
print(cx.shape,"\n",'-'*100)
print(c.shape,"\n",'-'*100)
print(c.dtype,"\n",'-'*100)
print(ax,"\n",'-'*100)
print(bx,"\n",'-'*100)
print(cx,"\n",'-'*100)
print(ax+bx*cx)###每个单独元素，互相相乘
#一共生成4*3*5=60个元素
result = ax+bx*cx
print(result[3,2,4])#对其进行索引,最后一个元素
```

# 对array的索引，单个元素的判断

## 输入

```python
import numpy as np

a = np.arange(12).reshape(3,4)
```

# 对numpy array的高级索引

**this is important！！！**

## 输入

```python
import numpy as np


a = np.arange(12)**2#生成1D array
print(a,"\n","-"*100)

i = np.array([1,1,3,8,5])#生成1D array
print(i,"\n","-"*100)

print(a[i],"\n","-"*100)#i作为参数传入a，不需要12，只要后面的乘方


#花式索引（Fancy indexing），利用整数数组进行索引

arr = np.zeros((8,4))
print(arr,"\n","-"*100)
for i in range(8):
	arr[i] = i

print(arr,"\n","-"*100)

#利用整数数组进行对arr的索引
print(arr[[4,3,0,6]],"\n","-"*100)
######传入一个数组，依次按数组中的数字，对arr进行索引
#使用负数，就会从末尾开始索引
print(arr[[-3,-5,-7]],"\n","-"*100)

print(arr[[1, 5, 7, 2], [0, 3, 1, 2]])
 # 打印arr[1, 0]、arr[5, 3]，arr[7, 1]和arr[2, 2]
 #用于对array中的单个元素的精确索引

```
